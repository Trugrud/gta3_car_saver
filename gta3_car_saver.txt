0000:

if and
    0256: player $PLAYER_CHAR defined
    8118: not actor $PLAYER_ACTOR dead
then
    1@ = False
    wait 1000 ms
    00BC: show_text_highpriority 'INIT' time 2000 flag 1
    
    wait 0
    while true
        wait 0
        if and 
            00E0: player $PLAYER_CHAR in_any_car 
            8118: not actor $PLAYER_ACTOR dead
        then
            if
                1@ == False
            then 
                00BC: show_text_highpriority 'IN_CAR' time 2000 flag 1
                1@ = True
            end
        else
            if and 
                80E0: not player $PLAYER_CHAR in_any_car
                8118: not actor $PLAYER_ACTOR dead
            then
                if 
                    1@ == True
                then
                    00BC: show_text_highpriority 'NIN_CAR' time 2000 flag 1
                    1@ = False
                end
            end
        end
    end
end


0A93: terminate_this_custom_script
